 15 var cg3d = (function() {
 16 
 17   var Matrix = lalgebra.Matrix;
 18   var errMsg = lalgebra.errMsg;
 19   var check2dArray = lalgebra.check2dArray;
 20 
 21   /**
 22    * Creates an object.
 23    * 
 24    * @memberof cg3d
 25    * @constructor
 26    * @this {Object3d}
 27    * @param {Array} vertices An array of elements for the Matrix.
 28    * @param {Array} edges An array of edges of the object.
 29    * @param {object} settings An object with properties of segments, points and captions.
 30    */
 31   function Object3d(vertices, edges, settings) {
 32     var cols = 0;
 33     var obj = new Matrix(vertices);
 34 
 35     if(obj.getCols()!==4) {
 36       throw errMsg("Matrix of the three-dimensional object should have four columns!", "cg-ob-001");
 37     }
 38     if(check2dArray(edges)) cols = edges[0].length;
 39 
 40     if(cols!==2) {
 41       throw errMsg("Matrix of edges should have two columns!", "cg-ob-002");
 42     }
 43     if(!(settings["segments"] && settings["points"] && settings["captions"])) {
 44       throw errMsg("Settings should have three properties: segments, points and captions!", "cg-ob-003");
 45     }
 46     /**
 47      * Copies the object.
 48      *
 49      * @memberof cg3d.Object3d
 50      * @return {Object3d} A copy of the object
 51      */
 52     this.getCopy = function() { 
 53       var els = obj.getElems();
 54       return new Object3d(els, edges, settings);
 55       //JSON.parse(JSON.stringify(edges)), JSON.parse(JSON.stringify(settings))
 56     };
 57     /**
 58      * Gets elements of the object matrix.
 59      *
 60      * @memberof cg3d.Object3d
 61      * @return {Array} An array of the object matrix
 62      */
 63     this.getElems = function() { 
 64       return obj.getElems();
 65     };
 66     /**
 67      * Applies a transformation to the object.
 68      *
 69      * @memberof cg3d.Object3d
 70      * @param {Transform3d} tMx Contains a matrix of the transformation.
 71      */
 72     this.applyTransformation = function(tMx) { 
 73       if(tMx instanceof Transform3d) { 
 74         obj = obj.multiply(tMx.getMatrix()).normolize();
 75       }
 76     };
 77     /**
 78      * Displays the array elements into columns like in matrixes.
 79      * @memberof cg3d.Object3d
 80      */
 81     this.trace = function() { obj.trace(); };
 82     /**
 83      * Returns coordinates of the object edges and their visual settings.
 84      *
 85      * @memberof cg3d.Object3d
 86      * @return {object} An object contains coordinates of the edges, their width and color.
 87      */
 88     this.getSegments = function() {
 89       var coords = [];
 90       var tmpArr = obj.getElems();
 91 
 92       var amountEdges = edges.length;
 93       var p1, p2;
 94 
 95       for(var i=0; i<amountEdges; i+=1) {
 96         p1 = tmpArr[edges[i][0]].slice(0, -1);
 97         p2 = tmpArr[edges[i][1]].slice(0, -1);
 98         coords.push({
 99           "p1":{"x":p1[0], "y":p1[1], "z":p1[2]},  
100           "p2":{"x":p2[0], "y":p2[1], "z":p2[2]}
101         });
102       }
103       return {
104         "coords": coords,
105         "width": settings.segments.width,
106         "color": settings.segments.color
107       };
108     };
109     /**
110      * Returns coordinates of the object verteces and their visual settings.
111      *
112      * @memberof cg3d.Object3d
113      * @return {object} An object contains coordinates of the verteces, their width and color.
114      */
115     this.getPoints = function() {
116       var coords = [];
117       var tmpArr = obj.getElems();
118       var rows = obj.getRows();
119       var cols = obj.getCols();
120       var p;
121 
122       for(var i=0; i<rows; i+=1) {
123         p = tmpArr[i].slice(0, -1);
124         coords.push({"x":p[0], "y":p[1], "z":p[2]});
125       }
126       return {
127         "coords": coords,
128         "width": settings.points.width,
129         "color": settings.points.color
130       };
131     };
132     /**
133      * Returns captions of the verteces and their visual settings.
134      *
135      * @memberof cg3d.Object3d
136      * @return {object} An object contains coordinates of the verteces, their texts, font and color.
137      */
138     this.getCaptions = function() {
139       var coords = this.getPoints().coords;
140 
141       return {
142         "coords": coords,
143         "texts": settings.captions.texts || [],
144         "font": settings.captions.font,
145         "color": settings.captions.color
146       };
147     };
148   }

150   /**
151    * Creates a matrix of combined transformations.
152    * 
153    * @memberof cg3d
154    * @constructor
155    * @this {Transform3d}
156    * @param {Array} els An array of elements for the transformation Matrix
157    */
158   function Transform3d(els) {
159     var elements = els ||  [[ 1,0,0,0 ],
160                             [ 0,1,0,0 ],
161                             [ 0,0,1,0 ],
162                             [ 0,0,0,1 ]];
163     var tMx = new Matrix(elements);
164     /**
165      * Displays the array elements into columns like in matrixes.
166      * @memberof cg3d.Transform3d
167      */
168     this.trace = function() { tMx.trace(); };
169     /**
170      * Returns a copy of the transformation Matrix.
171      *
172      * @memberof cg3d.Transform3d
173      * @return {Matrix} A copy of the transformation Matrix
174      */
175     this.getMatrix = function() { return tMx.getCopy(); };
176     /**
177      * Gets elements of the transformation Matrix.
178      *
179      * @memberof cg3d.Transform3d
180      * @return {Array} An array of the transformation Matrix
181      */
182     this.getElems = function() { return tMx.getElems(); };
183     /**
184      * Multiplies the matrix of combined transformations by translation matrix.
185      *
186      * @memberof cg3d.Transform3d
187      * @param {string} axis A name of axis.
188      * @param {number} v A value of the transformation.
189      */
190     this.move = function(axis, v) {
191       var tmpArr; 
192       
193       v = parseFloat(v);
194       if(isNaN(v)) axis = "n"; // function will return a unity martix
195 
196       switch(axis.toLowerCase())
197       {
198         case "z": tmpArr = [[ 1,0,0,0 ],
199                             [ 0,1,0,0 ],
200                             [ 0,0,1,0 ],
201                             [ 0,0,v,1 ]];
202         break;
203         case "x": tmpArr = [[ 1,0,0,0 ],
204                             [ 0,1,0,0 ],
205                             [ 0,0,1,0 ],
206                             [ v,0,0,1 ]];
207         break;
208         case "y": tmpArr = [[ 1,0,0,0 ],
209                             [ 0,1,0,0 ],
210                             [ 0,0,1,0 ],
211                             [ 0,v,0,1 ]];
212         break;
213         default:  tmpArr = [[ 1,0,0,0 ],
214                             [ 0,1,0,0 ],
215                             [ 0,0,1,0 ],
216                             [ 0,0,0,1 ]];
217       }
218       var transform = new Matrix(tmpArr);
219       tMx = tMx.multiply(transform);
220     };    
221     /**
222      * Multiplies the matrix of combined transformations by rotation matrix.
223      *
224      * @memberof cg3d.Transform3d
225      * @param {string} axis A name of axis.
226      * @param {number} angle An angle of rotation.
227      */
228     this.rotate = function(axis, angle) {
229       var tmpArr; 
230       var a;
231 
232       angle = parseFloat(angle);
233       if(isNaN(angle)) axis = "n"; // function will return a unity martix
234       else a = angle/180*Math.PI;
235 
236       switch(axis.toLowerCase())
237       {
238         case "z": tmpArr = [[ Math.cos(a), Math.sin(a), 0,  0],
239                             [-Math.sin(a), Math.cos(a), 0,  0],
240                             [ 0,           0,           1,  0],
241                             [ 0,           0,           0,  1]];
242           break;
243         case "x": tmpArr = [[1,      0,           0,          0],
244                             [0,      Math.cos(a), Math.sin(a),0],
245                             [0,     -Math.sin(a), Math.cos(a),0],
246                             [0,      0,           0,          1]];
247           break;
248         case "y": tmpArr = [[Math.cos(a), 0,     -Math.sin(a),0],
249                             [0,           1,      0,          0],
250                             [Math.sin(a), 0,      Math.cos(a),0],
251                             [0,           0,      0,          1]];
252           break;
253         default: tmpArr =  [[ 1,0,0,0 ],
254                             [ 0,1,0,0 ],
255                             [ 0,0,1,0 ],
256                             [ 0,0,0,1 ]];
257         }
258       var transform = new Matrix(tmpArr);
259       tMx = tMx.multiply(transform);
260     };
261     /**
262      * Multiplies the matrix of combined transformations by arbitary rotation matrix.
263      *
264      * @memberof cg3d.Transform3d
265      * @param {object} vector The arbitary axis.
266      * @param {number} angle An angle of rotation.
267      */
268     this.rotateFree = function(vector, angle) {
269       var crt = {};
270 
271       if(!lalgebra.checkSegment(vector)) {
272         crt = { x:0, y:0, z:1 };
273       }
274       else {
275         crt.x = vector.p2.x - vector.p1.x;
276         crt.y = vector.p2.y - vector.p1.y;
277         crt.z = vector.p2.z - vector.p1.z;
278       }
279       angle = parseFloat(angle);
280       if(isNaN(angle)) angle = 0;
281 
282       var sph = lalgebra.cartesianToSpherical(crt);
283       this.move("x", -vector.p1.x);
284       this.move("y", -vector.p1.y);
285       this.move("z", -vector.p1.z);
286       this.rotate("z", -sph.a);
287       this.rotate("y", -sph.b);
288       this.rotate("z", angle);
289       this.rotate("y", sph.b);
290       this.rotate("z", sph.a);
291       this.move("x", vector.p1.x);
292       this.move("y", vector.p1.y);
293       this.move("z", vector.p1.z);
294     }
295     /**
296      * Multiplies the matrix of combined transformations by scaling matrix.
297      *
298      * @memberof cg3d.Transform3d
299      * @param {string} axis A name of axis.
300      * @param {number} v A value of the transformation.
301      */
302     this.scale = function(axis, v) {
303       var tmpArr; 
304 
305       v = parseFloat(v);
306       if(isNaN(v)) axis = "n"; // function will return a unity martix
307 
308       switch(axis.toLowerCase())
309       {
310         case "z": tmpArr = [[ 1,0,0,0 ],
311                             [ 0,1,0,0 ],
312                             [ 0,0,v,0 ],
313                             [ 0,0,0,1 ]];
314         break;
315         case "x": tmpArr = [[ v,0,0,0 ],
316                             [ 0,1,0,0 ],
317                             [ 0,0,1,0 ],
318                             [ 0,0,0,1 ]];
319         break;
320         case "y": tmpArr = [[ 1,0,0,0 ],
321                             [ 0,v,0,0 ],
322                             [ 0,0,1,0 ],
323                             [ 0,0,0,1 ]];
324         break;
325         case "s": tmpArr = [[ 1,0,0,0 ],
326                             [ 0,1,0,0 ],
327                             [ 0,0,1,0 ],
328                             [ 0,0,0,v ]];
329         break;
330         default:  tmpArr = [[ 1,0,0,0 ],
331                             [ 0,1,0,0 ],
332                             [ 0,0,1,0 ],
333                             [ 0,0,0,1 ]];
334      }
335       var transform = new Matrix(tmpArr);
336       tMx = tMx.multiply(transform);
337     };
338     /**
339      * Multiplies the matrix of combined transformations by perspective matrix.
340      *
341      * @memberof cg3d.Transform3d
342      * @param {string} axis A name of axis.
343      * @param {number} v A value of the transformation.
344      */
345     this.perspective = function(axis, v) {
346       var tmpArr; 
347 
348       v = parseFloat(v);
349       if(isNaN(v)) axis = "n"; // function will return a unity martix
350 
351       switch(axis.toLowerCase())
352       {
353         case "z": tmpArr = [[ 1,0,0,0 ],
354                             [ 0,1,0,0 ],
355                             [ 0,0,1,v ],
356                             [ 0,0,0,1 ]];
357         break;
358         case "x": tmpArr = [[ 1,0,0,v ],
359                             [ 0,1,0,0 ],
360                             [ 0,0,1,0 ],
361                             [ 0,0,0,1 ]];
362         break;
363         case "y": tmpArr = [[ 1,0,0,0 ],
364                             [ 0,1,0,v ],
365                             [ 0,0,1,0 ],
366                             [ 0,0,0,1 ]];
367         break;
368         default:  tmpArr = [[ 1,0,0,0 ],
369                             [ 0,1,0,0 ],
370                             [ 0,0,1,0 ],
371                             [ 0,0,0,1 ]];
372       }
373       var transform = new Matrix(tmpArr);
374       tMx = tMx.multiply(transform);
375     };
376   }

378   /**
379    * Creates a scene with three-dimensional objects.
380    * 
381    * @memberof cg3d
382    * @constructor
383    * @this {Scene3d}
384    * @param {CanvasRenderingContext2D} ctx Two-dimensional context.
385    * @param {object} center A center of the canvas.
386    * @param {number} scale A koefficient of scaling.
387    */
388   function Scene3d(ctx, center, scale) {
389     var objects = [];
390     var view = new Transform3d;
391     var space = new Transform3d;
392 
393     if(!(ctx instanceof CanvasRenderingContext2D))
394       throw errMsg("Problem with a canvas context!", "cg-sc-001");
395 
396     center = center || {"x":0, "y":0};
397 
398     setSpace();
399     /**
400      * Creates a transformation matrix to convert object coordinates into the screen coordinates.
401      */
402     function setSpace() {
403       space = new Transform3d;
404 
405       space.scale("y", -1);
406       space.scale("s", 1/scale);
407       space.move("x", center.x);
408       space.move("y", center.y);
409     }
410     /**
411      * Removes one or more objects from the object array of the scene.
412      * 
413      * @memberof cg3d.Scene3d
414      * @param {number} idx An index of the object.
415      */
416     this.clearObjects = function(idx) {
417       idx = parseInt(idx);
418       if(isNaN(idx)) objects.splice(0);
419       else objects.splice(idx, 1);
420     }
421     /**
422      * Adds an object to the object array of the scene.
423      * 
424      * @memberof cg3d.Scene3d
425      * @param {Object3d} obj An three-dimensional object.
426      */
427     this.addObject = function(obj) {
428       objects.push(obj);
429     }
430     /**
431      * Displays the objects of the scene.
432      * 
433      * @memberof cg3d.Scene3d
434      */
435     this.draw = function() {
436       var obj, o;
437       ctx.clearRect(0, 0, ctx.canvas.width, ctx.canvas.height);
438       for(var i=0, count=objects.length; i<count; i+=1) {
439         obj = objects[i].getCopy();
440         obj.applyTransformation(view);
441         obj.applyTransformation(space);
442 
443         drawSegments(obj.getSegments());
444         drawPoints(obj.getPoints());
445         drawCaptions(obj.getCaptions());
446       }
447     }
448     /**
449      * Creates a transformation matrix to set a viewport up.
450      * 
451      * @memberof cg3d.Scene3d
452      * @param {object} vector A viewing vector.
453      * @param {boolean} isPerspective It's true if we needs a perspective view.
454      */
455     this.setView = function(vector, isPerspective) {
456       if(!lalgebra.checkSegment(vector)) return;
457 
458       var p1 = vector.p1, p2 = vector.p2;
459       var crt = {};
460 
461       crt.x = p2.x - p1.x;
462       crt.y = p2.y - p1.y;
463       crt.z = p2.z - p1.z;
464 
465       var mod = lalgebra.vectorMagnitude(crt);
466       if(mod===0) return;
467       
468       view = new Transform3d;
469 
470       var sph = lalgebra.getAngles(crt);
471       view.move("x", -p1.x);
472       view.move("y", -p1.y);
473       view.move("z", -p1.z);
474       view.rotate("y", -sph.a);
475       view.rotate("x", -sph.b+90);
476       if(isPerspective) view.perspective("z", -1/mod);
477     }
478     /**
479      * Displays the object edges.
480      *
481      * @memberof cg3d.Scene3d
482      */
483     function drawSegments(segments) {
484       var p1, p2;
485       var coords = segments.coords;
486 
487       ctx.beginPath();
488       for(var i=0, count=coords.length; i<count; i+=1) {
489         p1 = coords[i].p1;
490         p2 = coords[i].p2;
491         ctx.moveTo(p1.x, p1.y);
492         ctx.lineTo(p2.x, p2.y);
493       }
494       ctx.strokeStyle = segments.color;
495       ctx.lineWidth = segments.width;
496       ctx.stroke();
497     }
498     /**
499      * Displays the object verteces.
500      *
501      * @memberof cg3d.Scene3d
502      */
503     function drawPoints(points) {
504       var p;
505       var coords = points.coords;
506       var radius = points.width/2;
507 
508       ctx.fillStyle = points.color;
509       for(var i=0, count=coords.length; i<count; i+=1) {
510         p = coords[i];
511         ctx.beginPath();
512         ctx.moveTo(p.x, p.y);
513         ctx.arc(p.x, p.y, radius, 0, Math.PI*2, true);
514         ctx.fill();
515       }
516     }
517     /**
518      * Displays the captions of the verteces.
519      *
520      * @memberof cg3d.Scene3d
521      */
522     function drawCaptions(captions) {
523       var p;
524       var coords = captions.coords;
525       var texts = captions.texts;
526       ctx.font = captions.font;
527       ctx.fillStyle = captions.color;
528       ctx.textBaseline = "top";
529 
530       var count = (texts.length < coords.length)? texts.length : coords.length;
531       for(var i=0; i<count; i+=1) {
532         p = coords[i];
533         ctx.beginPath();
534         ctx.fillText(texts[i], p.x+2, p.y+2);
535         ctx.fill();
536       }
537     }
538   }

544   }
